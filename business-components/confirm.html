Confirm 二次确认弹窗 ​

基本用法 ​

支持hooks和函数式调用，推荐在组合式中使用hooks方式。

点我查看代码
vue
<script setup lang="ts">
import { confirm, useConfirm } from '@/components/Confirm';
/** hooks */
import { createVNode } from 'vue';
const { confirm: confirmHook } = useConfirm();
/** confirm hooks */
const confirmHooks = () => {
  confirmHook({
    title: 'Do you Want to delete these items?',
    content: createVNode('div', { style: 'color:red;' }, 'Some descriptions'),
    onOk() {
      console.log('OK');
    },
    onCancel() {
      console.log('Cancel');
    },
    class: 'test',
  });
};

/** confirm function */
const confirmFn = () => {
  confirm({
    title: 'Do you Want to delete these items?',
    content: createVNode('div', { style: 'color:red;' }, 'Some descriptions'),
    onOk() {
      console.log('OK');
    },
    onCancel() {
      console.log('Cancel');
    },
    class: 'test',
  });
};
</script>
<template>
  <a-space class="demo1">
    <a-button type="primary" ghost @click="confirmHooks">Hooks Comfirm</a-button>
    <a-button type="primary" ghost @click="confirmFn">Function Comfirm</a-button>
  </a-space>
</template>

<style lang="less" scoped>
.demo1 {
  margin-top: 20px;
}
</style>

其他提示类型 ​

包括警告(default)、警告丿、消息提示。

点我查看代码
vue
<script setup lang="ts">
import { useConfirm } from '@/components/Confirm';
/** hooks */
import { createVNode } from 'vue';

const { confirm, warning, info } = useConfirm();
/** confirm hooks */
const confirmForDefault = () => {
  confirm({
    title: 'Do you Want to delete these items?',
    content: createVNode('div', { style: 'color:red;' }, 'Some descriptions'),
    onOk() {
      console.log('OK');
    },
    onCancel() {
      console.log('Cancel');
    },
    class: 'test',
  });
};
const confirmForWarning = () => {
  warning({
    title: 'Do you Want to delete these items?',
    content: createVNode('div', { style: 'color:red;' }, 'Some descriptions'),
    onOk() {
      console.log('OK');
    },
    onCancel() {
      console.log('Cancel');
    },
    class: 'test',
  });
};
const confirmForInfo = () => {
  info({
    title: 'Do you Want to delete these items?',
    content: createVNode('div', { style: 'color:red;' }, 'Some descriptions'),
    onOk() {
      console.log('OK');
    },
    onCancel() {
      console.log('Cancel');
    },
    class: 'test',
  });
};
</script>
<template>
  <a-space class="demo2">
    <a-button type="primary" ghost @click="confirmForDefault">Default</a-button>
    <a-button type="primary" ghost @click="confirmForWarning">Warning</a-button>
    <a-button type="primary" ghost @click="confirmForInfo">Info</a-button>
  </a-space>
</template>

  margin-top: 20px;
}
</style>

异步关闭 ​

支持Promise 异步操作。
⭐ 新增 - modal.confirm({....}).then(res => {}) .then 用法，扩展异步确认后的回调，来触发刷新页面、其他操作。

点我查看代码
vue
<script setup lang="ts">
import { confirm, useConfirm } from '@/components/Confirm';
const { confirm: confirmHook } = useConfirm();
/** confirm hooks */
const confirmHooks = () => {
  confirmHook({
    title: 'Do you want to delete these items?',
    content: 'When clicked the OK button, this dialog will be closed after 1 second',
    async onOk() {
      return await new Promise((resolve, reject) => {
        setTimeout(Math.random() > 0.5 ? resolve : reject, 1000);
      });
    },
    onCancel() {},
  })
    .then(() => {
      console.log('refresh page');
    })
    .catch(console.error);
};

/** confirm function */
const confirmFn = () => {
  confirm({
    title: 'Do you want to delete these items?',
    content: 'When clicked the OK button, this dialog will be closed after 1 second',
    async onOk() {
      return await new Promise((resolve, reject) => {
        setTimeout(Math.random() > 0.5 ? resolve : reject, 1000);
      });
    },
    onCancel() {},
  })
    .then(() => {
      console.log('refresh page');
    })
    .catch(console.error);
};
</script>
<template>
  <a-space class="demo3">
    <a-button type="primary" ghost @click="confirmHooks">Hooks Promise Comfirm</a-button>
    <a-button type="primary" ghost @click="confirmFn">Function Promise Comfirm</a-button>
  </a-space>
</template>

<style lang="less" scoped>
.demo3 {
  margin-top: 20px;
}
</style>
